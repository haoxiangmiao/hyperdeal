50  template <bool forward, bool add>
51  void
52  interpolate(Number * output,
53  const Number * input,
54  const unsigned int face_no)
55  {
56  // clang-format off
57  if (dim >= 1 && face_no / 2 == 0) interpolate_impl<0, forward, add>(output, input, face_no); else
58  if (dim >= 2 && face_no / 2 == 1) interpolate_impl<1, forward, add>(output, input, face_no); else
59  if (dim >= 3 && face_no / 2 == 2) interpolate_impl<2, forward, add>(output, input, face_no); else
60  if (dim >= 4 && face_no / 2 == 3) interpolate_impl<3, forward, add>(output, input, face_no); else
61  if (dim >= 5 && face_no / 2 == 4) interpolate_impl<4, forward, add>(output, input, face_no); else
62  if (dim >= 6 && face_no / 2 == 5) interpolate_impl<5, forward, add>(output, input, face_no); else
63  {
64  Assert(false, dealii::StandardExceptions::ExcNotImplemented());
65  }
66  // clang-format on
67  }

73  template <int face_direction, bool contract_onto_face, bool add>
74  void
75  interpolate_impl(Number *DEAL_II_RESTRICT out,
76  const Number *DEAL_II_RESTRICT in,
77  const unsigned int face_no) const
78  {
79  AssertIndexRange(face_direction, dim);
80 
81  constexpr auto n_blocks1 =
82  dealii::Utilities::pow<unsigned int>(n_rows, face_direction);
83  constexpr auto n_blocks2 = dealii::Utilities::pow<unsigned int>(
84  n_rows, std::max(dim - face_direction - 1, 0));
85 
86  constexpr auto stride =
87  dealii::Utilities::pow<unsigned int>(n_rows, face_direction);
88 
89  const Number *DEAL_II_RESTRICT shape_values =
90  &shape_info.data[0].quadrature_data_on_face[face_no % 2][0];
91 
92  for (unsigned int i2 = 0u; i2 < n_blocks2; ++i2)
93  {
94  for (unsigned int i1 = 0u; i1 < n_blocks1; ++i1)
95  {
96  if (contract_onto_face)
97  {
98  Number res0 = in[0] * shape_values[0];
99 
100  for (unsigned int ind = 1; ind < n_rows; ++ind)
101  res0 += in[ind * stride] * shape_values[ind];
102 
103  if (add == false)
104  out[0] = res0;
105  else
106  out[0] += res0;
107  }
108  else
109  {
110  for (unsigned int col = 0; col < n_rows; ++col)
111  {
112  if (add == false)
113  out[col * stride] = in[0] * shape_values[col];
114  else
115  out[col * stride] += in[0] * shape_values[col];
116  }
117  }
118 
119  ++out;
120  ++in;
121  }
122 
123  if (contract_onto_face)
124  in += (dealii::Utilities::pow(n_rows, face_direction + 1) -
125  n_blocks1);
126  else
127  out += (dealii::Utilities::pow(n_rows, face_direction + 1) -
128  n_blocks1);
129  }
130  }
131 
132  // TODO: use the hyper.deal version?
133  const dealii::internal::MatrixFreeFunctions::ShapeInfo<Number>
134  &shape_info;
135  };
