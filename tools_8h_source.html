39  template <int dim, typename AdditionalData>
40  void
41  categorize_accoring_boundary_ids_for_ecl(
42  const Triangulation<dim> &tria,
43  AdditionalData & additional_data,
44  const unsigned int level = numbers::invalid_unsigned_int)
45  {
46  const bool is_mg = (level != numbers::invalid_unsigned_int);
47 
48  // ... create list for the category of each cell
49  if (is_mg)
50  additional_data.cell_vectorization_category.resize(
51  std::distance(tria.begin(level), tria.end(level)));
52  else
53  additional_data.cell_vectorization_category.resize(
54  tria.n_active_cells());
55 
56  // ... setup scaling factor
57  std::vector<unsigned int> factors(dim * 2);
58 
59  std::map<unsigned int, unsigned int> bid_map;
60  for (unsigned int i = 0; i < tria.get_boundary_ids().size(); i++)
61  bid_map[tria.get_boundary_ids()[i]] = i + 1;
62 
63  {
64  unsigned int bids = tria.get_boundary_ids().size() + 1;
65  int offset = 1;
66  for (unsigned int i = 0; i < dim * 2; i++, offset = offset * bids)
67  factors[i] = offset;
68  }
69 
70  const auto to_category = [&](auto &cell) {
71  unsigned int c_num = 0;
72  for (unsigned int i = 0; i < dim * 2; i++)
73  {
74  auto &face = *cell->face(i);
75  if (face.at_boundary())
76  c_num += factors[i] * bid_map[face.boundary_id()];
77  }
78  return c_num;
79  };
80 
81  if (!is_mg)
82  {
83  for (const auto &cell : tria.active_cell_iterators())
84  {
85  if (cell->is_locally_owned())
86  additional_data
87  .cell_vectorization_category[cell->active_cell_index()] =
88  to_category(cell);
89  }
90  }
91  else
92  {
93  for (const auto &cell : tria.cell_iterators_on_level(level))
94  {
95  if (cell->is_locally_owned_on_level())
96  additional_data.cell_vectorization_category[cell->index()] =
97  to_category(cell);
98  }
99  }
100 
101  // ... finalize setup of matrix_free
102  additional_data.hold_all_faces_to_owned_cells = true;
103  additional_data.cell_vectorization_categories_strict = true;
104  additional_data.mapping_update_flags_faces_by_cells =
105  additional_data.mapping_update_flags_inner_faces |
106  additional_data.mapping_update_flags_boundary_faces;
107  }

110  template <int dim, typename VectorizedArrayType>
111  VectorizedArrayType
112  evaluate_scalar_function(
113  const dealii::Point<dim, VectorizedArrayType> & point,
114  const Function<dim, typename VectorizedArrayType::value_type> &function,
115  const unsigned int n_lanes)
116  {
117  VectorizedArrayType result;
118 
119  for (unsigned int v = 0; v < n_lanes; ++v)
120  {
121  dealii::Point<dim> p;
122  for (unsigned int d = 0; d < dim; ++d)
123  p[d] = point[d][v];
124  result[v] = function.value(p);
125  }
126 
127  return result;
128  }
